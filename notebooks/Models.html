def build_model(hp):
    model = Sequential()

    # First dense layer: units 64512
    model.add(Dense(
        units=hp.Int('units_1', min_value=64, max_value=512, step=64),
        activation='relu',
        input_dim=input_dim
    ))
    model.add(BatchNormalization())
    model.add(Dropout(hp.Float('dropout_1', min_value=0.0, max_value=0.5, step=0.1)))

    # Optionally a second hidden layer
    if hp.Boolean('use_second_layer'):
        model.add(Dense(
            units=hp.Int('units_2', min_value=32, max_value=256, step=32),
            activation='relu'
        ))
        model.add(Dropout(hp.Float('dropout_2', min_value=0.0, max_value=0.5, step=0.1)))

    # Optional third layer
    if hp.Boolean('use_third_layer'):
        model.add(Dense(
            units=hp.Int('units_3', min_value=16, max_value=128, step=16),
            activation='relu'
        ))
        model.add(Dropout(hp.Float('dropout_3', min_value=0.0, max_value=0.5, step=0.1)))
    # Optional fourth layer
    if hp.Boolean('use_fourth_layer'):
        model.add(Dense(
            units=hp.Int('units_4', min_value=8, max_value=64, step=8),
            activation='relu'
        ))
        model.add(Dropout(hp.Float('dropout_3', min_value=0.0, max_value=0.5, step=0.1)))

    # Output layer
    model.add(Dense(1, activation='sigmoid'))

    # Tune learning rate
    lr = hp.Choice('learning_rate', values=[1e-2, 5e-3, 1e-3, 5e-4])
    optimizer = tf.keras.optimizers.Adam(learning_rate=lr)

    model.compile(
        optimizer=optimizer,
        loss='binary_crossentropy',
        metrics=['accuracy']
    )

    return model

from xgboost import XGBClassifier
from sklearn.metrics import classification_report, roc_auc_score, fbeta_score, make_scorer
from sklearn.model_selection import GridSearchCV, StratifiedKFold

# --- 1. Define a More Robust Model (Regularized) ---
# We reduce max_depth and increase min_child_weight to prevent overfitting to the minority class
xgb_robust = XGBClassifier(
    n_estimators=150,
    learning_rate=0.05,        # Slower learning for better generalization
    max_depth=3,               # Shallower trees prevent memorizing noise
    min_child_weight=3,        # Requires more samples to create a leaf (reduces outliers)
    scale_pos_weight=3.5,      # Keep the aggressive weighting
    subsample=0.8,             # Use only 80% of data per tree to prevent overfitting
    colsample_bytree=0.8,
    random_state=42,
    eval_metric='logloss'
)

# Train the robust model
print("Training Robust XGBoost...")
xgb_robust.fit(X_train, y_train)

# Get probabilities
y_proba_robust = xgb_robust.predict_proba(X_test)[:, 1]

# --- 2. Threshold Tuning for F2-SCORE (Focus on Recall) ---
# F2 score weights Recall higher than Precision. This is crucial for Loan Default.

thresholds = np.arange(0.01, 0.57, 0.01)
f2_scores = [fbeta_score(y_test, y_proba_robust >= t, beta=2) for t in thresholds]

# Find the threshold that gives the best F2 Score
optimal_threshold_f2 = thresholds[np.argmax(f2_scores)]
max_f2 = np.max(f2_scores)

print(f"\n--- Threshold Tuning (Optimizing for F2-Score) ---")
print(f"Optimal Threshold: {optimal_threshold_f2:.3f}")
print(f"Max F2-Score: {max_f2:.4f}")

# Apply the new threshold
y_pred_f2 = (y_proba_robust >= optimal_threshold_f2).astype(int)

# --- 3. Final Evaluation ---
print("\n--- Final Results (Optimized for Recall) ---")
print(classification_report(y_test, y_pred_f2))

# Calculate AUC
auc_robust = roc_auc_score(y_test, y_proba_robust)
recall_robust = classification_report(y_test, y_pred_f2, output_dict=True)['1']['recall']
